39 /**
40  * Asynchronously sends HTTP requests to etcd servers and handles outcome responses.
41  *
42  * ==EtcdClient==
43  *
44  * EtcdClient uses [[https://github.com/AsyncHttpClient/async-http-client/ AsyncHttpClient]] in order to build and send HTTP requests.
45  * Its methods were developed to suit the etcd [[https://coreos.com/etcd/docs/latest/ documentation]].
46  * All methods use [[http://docs.scala-lang.org/overviews/core/futures.html Scala Futures and Promises]] in order to provide an asynchronous implementation.
47  *
48  * Depending on the request, either an error will be thrown or a response will be returned.
49  * In both cases, there is a case class extending a trait in package [[com.eiipii.etcd.client.model model]] with a name ending in Result, which will be returned containing either the response or information about the error.
50  *
51  * For example, when performing an [[com.eiipii.etcd.client.EtcdClient.setKey EtcdClient.setKey]] operation, an [[com.eiipii.etcd.client.model.EtcdSetKeyResult EtcdSetKeyResult]] is expected.
52  * The outcome of the operation, either an error or a response, can be accessed considering both cases.
53  * An easy way to retrieve a response could be with a `match` statement:
54  *
55  * {{{
56  *        etcdcli.setKey(key, value) onSuccess  {
57  *        case EtcdSetKeyResponse(headers, body) =>
58  *         ???
59  *        }
60  * }}}
61  *
62  * ===Key-Value Methods===
63  *
64  * The implementation of key-value methods follows the specification given in the [[https://coreos.com/etcd/docs/latest/v2/api.html etcd Client API]].
65  * Their purpose is to set, retrieve, modify or delete keys, values and directories in etcd.
66  * It is also possible to wait (set a watch) on etcd keys, so as to monitor changes in the server in real time and conditionally set operations depending on the response obtained once a change in the data is registered.
67  *
68  * Key-Value methods take as inputs case classes defined in package [[model]].
69  * The recommended way of getting input data ([[com.eiipii.etcd.client.model.EtcdKey EtcdKey]]s, [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]]s, hidden nodes and [[com.eiipii.etcd.client.model.EtcdValue EtcdValue]]s) for key-value methods is using the [[com.eiipii.etcd.client.model.EtcdModel EtcdModel]] object.
70  *
71  * ===Statistics===
72  *
73  * The implementation of methods for retrieving etcd cluster statistics also follows the specification given in the [[https://coreos.com/etcd/docs/latest/v2/api.html etcd Client API]] and [[https://coreos.com/etcd/docs/latest/v2/other_apis.html etcd Admin API]].
74  *
75  * ===Authentication Methods===
76  *
77  * The implementation of authentication methods follows the specification given in the [[https://coreos.com/etcd/docs/latest/v2/auth_api.html etcd authentication and security API]].
78  * With the help of these methods, a system administrator can enable authentication and create users and roles with different levels of access to etcd.
79  * etcd only allows authentication on authentication methods and key-value methods.
80  *
81  * ===Member Methods===
82  *
83  * The implementation for member methods follows the specification given in the [[https://coreos.com/etcd/docs/latest/v2/members_api.html etcd  Members API]].
84  * They allow an administrator to configure a cluster, by changing the members of a cluster.
85  *
86  * @param etcdConfig [[com.eiipii.etcd.client.EtcdConfiguration EtcdConfiguration]] with information about an endpoint to which requests will be sent and optional authorization data.
87  *
88  */
89

103   /**
104    * Create a new EtcdClient instance with support for TLS.
105    *
106    * See the [[https://coreos.com/etcd/docs/latest/v2/security.html V2 Security Model]] documentation in order to learn about the possible use cases of etcd with TLS.
107    *
108    * The following is an example of a possible configuration with TLS taken from out tests:
109    *
110    * {{{
111    *  val clientCert = this.getClass.getResourceAsStream("/ssl/client.crt")
112    *  val keyFile = this.getClass.getResourceAsStream("/ssl/client_nopass.PKCS8")
113    *  val caCert = this.getClass.getResourceAsStream("/ssl/dev-ca.crt")
114    *  val sslContext = SslContextBuilder.forClient()
115    *  .trustManager(caCert)
116    *  .keyManager(clientCert, keyFile)
117    *  .build()
118    *
119    *  val endpoint: String = s"https://localhost:4001"
120    *  val etcdCli = new EtcdClient(EtcdDSL.config(endpoint))
121    *  val secureCli: EtcdClient = etcdCli.withTLS(sslContext)
122    * }}}
123    *
124    *  Make sure to fill in the details for properly configuring the TLS protocol according to your needs.
125    * @param nettySslContext Look for io.netty.handler.ssl.SslContext in the [[http://netty.io/4.0/api/index.html Netty Javadoc]].
126    */
127 
128   def withTLS(nettySslContext: SslContext): EtcdClient = EtcdClient(etcdConfig.withTLS(nettySslContext))
129   /**
130    * Close the internal resources.
131    */
132   def close(): Unit = asyncHttpClient.close()

134   /*
135   Uses DefaultAsyncHttpClientConfig and the parameter etcdConfig in order to configure asyncHttpClient used internally for http requests to etcd.
136    */
137   private lazy val constructAsyncHttpConfiguration: DefaultAsyncHttpClientConfig = {
138     import Dsl._
139     val baseBuilder = new DefaultAsyncHttpClientConfig.Builder()
140       .setKeepAlive(true)
141     val postAuth = etcdConfig.authorization match {
142       case Some(x) => baseBuilder.setRealm(
143         basicAuthRealm(x.userLogin, x.password).setUsePreemptiveAuth(true)
144       )
145       case None => baseBuilder
146     }
147 
148     val configTLS = etcdConfig.optContext match {
149       case Some(nettySslContext) =>
150         postAuth.setSslContext(nettySslContext)
151       case None =>
152         postAuth
153     }
154     configTLS.build()
155   }

173   /**
174    * Sets or updates a key in a specific etcd node; also, sets and unsets TTL.
175    *
176    * ===Key-Value Method===
177    *
178    * Implemented according to etcd Client API: "[[https://coreos.com/etcd/docs/latest/v2/api.html#setting-the-value-of-a-key Setting the value of a key]]".
179    * See also "[[https://coreos.com/etcd/docs/latest/v2/api.html#using-key-ttl Using key TTL]]", "[[https://coreos.com/etcd/docs/latest/v2/api.html#creating-a-hidden-node Creating a hidden node]]" and "[[https://coreos.com/etcd/docs/latest/v2/api.html#atomic-compare-and-swap Atomic Compare-and-Swap]]".
180    *
181    * The value of a key can be updated by choosing the same key, but a different value.
182    *
183    * Optionally one can choose a TTL (time to live), which is a lifespan in seconds for the key-value.
184    * When it expires, the key is deleted.
185    *
186    * It is also possible to update the TTL of a key by specifying a new value for the ttlOp parameter with the same key.
187    * A similar effect can be achieved using [[com.eiipii.etcd.client.EtcdClient.refreshTTL EtcdClient.refreshTTL]], except that this last operation does not notify watchers.
188    *
189    * The TTL can be unset by updating the key with the parameter ttlOp set to None and with condition KeyMustExist set to true.
190    *
191    * Additionally, etcd allows to conditionally compare and swap values.
192    * This can be done by setting the conditionOp parameter to any of the case classes extending trait [[com.eiipii.etcd.client.model.CompareAndSwapCondition CompareAndSwapCondition]].
193    *
194    * @param key [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] that is to be set or updated.
195    * @param value [[com.eiipii.etcd.client.model.EtcdValue EtcdValue]] to be associated to key.
196    * @param ttlOp Life span in seconds of key-value.
197    *              Defaults to None.
198    * @param conditionOp One of case classes [[com.eiipii.etcd.client.model.KeyMustExist KeyMustExist]], [[com.eiipii.etcd.client.model.KeyMustHaveIndex KeyMustHaveIndex]] or [[com.eiipii.etcd.client.model.KeyMustHaveValue KeyMustHaveValue ]] extending [[com.eiipii.etcd.client.model.CompareAndSwapCondition CompareAndSwapCondition]].
199    *                    Defaults to None.
200    * @return A Future wrapped around either [[com.eiipii.etcd.client.model.EtcdSetKeyResponse EtcdSetKeyResponse]] or [[com.eiipii.etcd.client.model.EtcdRequestError EtcdRequestError]] extending [[com.eiipii.etcd.client.model.EtcdSetKeyResult EtcdSetKeyResult]].
201    */
202   def setKey(key: EtcdKey, value: EtcdValue, ttlOp: Option[EtcdTTL] = None, conditionOp: Option[CompareAndSwapCondition] = None): Future[EtcdSetKeyResult] = {
203     val baseRequest = asyncHttpClient.preparePut(baseUrl + key.key).addFormParam("value", value.value)
204     build(baseRequest)(Seq(addSwapCondition(conditionOp), addTTL(ttlOp)))
205   }

207   /**
208    * Sets a watch, which returns a get response if there is a change in a key in etcd.
209    *
210    * ===Key-Value Method===
211    *
212    * Implemented according to etcd Client API: "[[https://coreos.com/etcd/docs/latest/v2/api.html#waiting-for-a-change Waiting for a change]]".
213    *
214    * Use case of the waitForKey method:
215    *
216    *
217    * Optionally, one can choose an etcd modified index for which the watch is being set.
218    * An etcd modified index is the integer associated to the operation modifying the state of a key in etcd.
219    * If an index `i` is chosen, such that there were subsequent changes in the state of the key after the operation associated to `i`, etcd returns an [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] with the state of the key after the first change registered with indexe greater than `i`.
220    *
221    * For an index `j` greater than the etcd index associated to the last operation performed, the watch will not return a response until a change with an index greater than `j` is made.
222    *
223    * Also, one can choose to wait for changes in any of the subnodes of a given node.
224    * This can be achieved by setting the parameter recursive to true.
225    * This parameter defaults to false.
226    *
227    * @param key                  [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] on watch.
228    * @param callback             [[com.eiipii.etcd.client.model.EtcdWaitCallback EtcdWaitCallback]] which will return a [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] or a throwable once it is called upon.
229    * @param modifiedIndexForWait [[com.eiipii.etcd.client.model.EtcdIndex EtcdIndex]] associated to an operation performed on a given key.
230    * @param recursive            waits on changes of subnoded of the specified key, when it is set to true.
231    *                             Defaults to false.
232    * @return A Future wrapped around either [[com.eiipii.etcd.client.model.EtcdWaitForKeyAccepted]] containing the headers of the response  or [[com.eiipii.etcd.client.model.EtcdRequestError EtcdRequestError]].
233    */
234   def waitForKey(key: EtcdKey, callback: EtcdWaitCallback, modifiedIndexForWait: Option[EtcdIndex] = None, recursive: Boolean = false): Future[EtcdWaitForKeyResult] = {
235     val baseRequest = asyncHttpClient.prepareGet(baseUrl + key.key)
236     (
237       build(baseRequest)(Seq(addWait(), addWaitIndex(modifiedIndexForWait), addRecursive(recursive))),
238       callback
239     )
240   }

242   /**
243    * Gets the current state of a key in etcd.
244    *
245    * ===Key-Value Method===
246    *
247    * Implemented according to etcd Client API: "[[https://coreos.com/etcd/docs/latest/v2/api.html#get-the-value-of-a-key Get the value of a key]]".
248    *
249    * Note that there is no difference in a GET request for a key and a directory.
250    * An GET HTTP request made to a node used as a directory looks the same as a request to a node used as a key.
251    * However, there is a difference in terms of a response.
252    * The getKey method is intended to handle responses only for requests to nodes used as keys.
253    * If it is used to make a request to a directory, it will return an EtcdRequestError with the statusCode field set to 200.
254    * This means that etcd accepts the request and returns a response.
255    * However EtcdClient does not parse the response.
256    * In order to get a response which extracts the JSON returned in the response body into a case class, use [[com.eiipii.etcd.client.EtcdClient.listDir EtcdClient.listDir]].
257    *
258    * @param key [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] whose status is queried.
259    * @return A Future wrapped around either [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] or [[com.eiipii.etcd.client.model.EtcdRequestError EtcdRequestError]] extending [[com.eiipii.etcd.client.model.EtcdGetKeyResult EtcdGetKeyResult]]
260    */
261   def getKey(key: EtcdKey): Future[EtcdGetKeyResult] = {
262     asyncHttpClient.prepareGet(baseUrl + key.key)
263   }

354   /**
355    * Updates the TTL of a key without notifying watchers.
356    *
357    * ===Key-Value Method===
358    *
359    * Implemented according to etcd Client API: "[[https://coreos.com/etcd/docs/latest/v2/api.html#refreshing-key-ttl Refreshing key TTL]]".
360    *
361    * When a watch is set, and a key is updated using [[com.eiipii.etcd.client.EtcdClient.setKey EtcdClient.setKey]], watchers are notified, triggering any operation conditionally set to follow given a particular response.
362    * This includes updates to its TTL.
363    * etcd allows updating TTL without notifying watches set on a key, thus not causing such side effects.
364    * EtcdClient implements this feature in the refreshTTL method.
365    * When a TTL is refreshed, its value cannot be updated.
366    *
367    * @param key [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] that is to be updated.
368    * @param ttlOp Life span in seconds of key-value .
369    *              Defaults to None.
370    * @param conditionOp One of case classes [[com.eiipii.etcd.client.model.KeyMustExist KeyMustExist]], [[com.eiipii.etcd.client.model.KeyMustHaveIndex KeyMustHaveIndex]] or [[com.eiipii.etcd.client.model.KeyMustHaveValue KeyMustHaveValue ]] extending [[com.eiipii.etcd.client.model.CompareAndSwapCondition CompareAndSwapCondition]].
371    *                    Defaults to None.
372    * @return A Future wrapped around either [[com.eiipii.etcd.client.model.EtcdSetKeyResponse EtcdSetKeyResponse]] or [[com.eiipii.etcd.client.model.EtcdRequestError EtcdRequestError]] extending [[com.eiipii.etcd.client.model.EtcdSetKeyResult EtcdSetKeyResult]].
373    */
374   def refreshTTL(key: EtcdKey, ttlOp: EtcdTTL, conditionOp: Option[CompareAndSwapCondition] = None): Future[EtcdSetKeyResult] = {
375     val baseRequest = asyncHttpClient.preparePut(baseUrl + key.key).addFormParam("ttl", ttlOp.ttl.toString).addFormParam("refresh", "true")
376     build(baseRequest)(Seq(addSwapCondition(conditionOp)))
377   }

400   /**
401    * Creates a key in a specified directory whose name is the current etcd index preceded with leading zeroes.
402    *
403    * ===Key-Value Method===
404    *
405    * Implemented according to etcd Client API: "[[https://coreos.com/etcd/docs/latest/v2/api.html#atomically-creating-in-order-keys Atomically Creating In-Order Keys]]".
406    *
407    * etcd keeps a counter for each operation performed.
408    * The etcd index keeps track of this counter, that is, it is the integer associated to the last operation performed in etcd.
409    * Given a specified directory [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]], createKeyFromCurrentEtcdIndex creates a key in that directory named after the etcd index and associates the value inputed through parameter value to it.
410    * This allows to create keys which are strictly ordered by those indexes, since etcd indexes increment monotonously.

857 /*
858 * Optional parameters for requests are also dealt with using implicits.
859 * This is achieved using the methods defined in the n EtcdClientParamatersCreation private object.
860 */
861 
862 private object EtcdClientParamatersCreation {
863   type EtcdApiParameterFunction = BoundRequestBuilder => BoundRequestBuilder
864 
865   def addSwapCondition(conditionOp: Option[CompareAndSwapCondition]): EtcdApiParameterFunction = builder => {
866     conditionOp match {
867       case Some(condition) => condition match {
868         case KeyMustExist(state) => builder.addFormParam("prevExist", if (state) "true" else "false")
869         case KeyMustHaveIndex(index) => builder.addFormParam("prevIndex", index.index.toString)
870         case KeyMustHaveValue(value) => builder.addFormParam("prevValue", value.value)
871       }
872       case None => builder
873     }
874   }
875 
876   def addDeleteCondition(deleteConditionOp: Option[ConditionalDeleteCondition]): EtcdApiParameterFunction = builder => {
877     deleteConditionOp match {
878       case Some(KeyMustHaveIndex(index)) => builder.addQueryParam("prevIndex", index.index.toString)
879       case Some(KeyMustHaveValue(value)) => builder.addQueryParam("prevValue", value.value)
880       case None => builder
881     }
882   }
883 
884   def addTTL(ttlOp: Option[EtcdTTL]): EtcdApiParameterFunction = builder => {
885     ttlOp match {
886       case Some(ttl) => builder.addFormParam("ttl", ttl.ttl.toString)
887       case None => builder
888     }
889   }
890 
891   def addWait(): EtcdApiParameterFunction = builder => {
892     builder.addQueryParam("wait", "true")
893   }

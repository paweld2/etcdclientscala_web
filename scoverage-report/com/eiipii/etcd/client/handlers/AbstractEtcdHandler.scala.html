30 abstract class AbstractEtcdHandler[T] extends AsyncHandler[Unit] {
31 
32   protected[this] var statusCodeOp: Option[Int] = None
33   protected[this] val bytes: ByteArrayOutputStream = new ByteArrayOutputStream()
34 
35   def promise: Promise[T]
36 
37   private lazy val logger: Logger = LoggerFactory.getLogger(this.getClass)
38 
39   def handlerName: String = this.getClass.getSimpleName
40 
41   override def onCompleted(): Unit = {
42     val result: Option[T] = statusCodeOp match {
43       case Some(ok) if ok < 300 => buildResponse()
44       case Some(error) => buildError()
45       case None => None //will result on a exception
46     }
47     result match {
48       case Some(res) => promise.success(res)
49       case None => failPromise()
50     }
51   }
52 
53   override def onThrowable(t: Throwable): Unit = {
54     promise.failure(t)
55     logger.debug(t.getMessage(), t)
56   }
57 
58   override def onBodyPartReceived(bodyPart: HttpResponseBodyPart): State = {
59     bytes.write(bodyPart.getBodyPartBytes())
60     State.CONTINUE
61   }
62 
63   override def onStatusReceived(responseStatus: HttpResponseStatus): State = {
64     val code: Int = responseStatus.getStatusCode
65     statusCodeOp = Some(code)
66     State.CONTINUE
67   }
68 
69   protected def buildResponse(): Option[T]
70 
71   def failPromise(): Unit
72 
73   def buildError(): Option[T]
74 
75 }

143 abstract class AbstractEtcdHandlerWithCompleteEtcdHeader[T >: EtcdRequestError] extends AbstractEtcdHandler[T] {
144 
145   def createResponse(status: Int, header: EtcdHeaders, json: String): Option[T]
146 
147   override def failPromise(): Unit = promise.failure(
148     IncorrectEtcdRequestFormatException(
149       handlerName,
150       statusCodeOp,
151       Some(bytes.toString("UTF-8")),
152       headsOp
153     )
154   )
155 
156   protected var headsOp: Option[EtcdHeaders] = None
157 
158   override def onHeadersReceived(headers: HttpResponseHeaders): State = {
159     headsOp = for {
160       clusterId <- Option(headers.getHeaders.get("X-Etcd-Cluster-Id")).map(EtcdClusterId)
161       etcdIndex <- Option(headers.getHeaders.get("X-Etcd-Index")).map(_.toInt).map(EtcdIndex)
162       raftIndex <- Option(headers.getHeaders.get("X-Raft-Index")).map(_.toInt).map(RaftIndex)
163       raftTerm <- Option(headers.getHeaders.get("X-Raft-Term")).map(_.toInt).map(RaftTerm)
164     } yield EtcdHeaders(clusterId, etcdIndex, raftIndex, raftTerm)
165     State.CONTINUE
166   }
167 
168   def buildResponse(): Option[T] = for {
169     status <- statusCodeOp
170     header <- headsOp
171     t <- createResponse(status, header, bytes.toString("UTF-8"))
172   } yield t
173 
174   override final def buildError(): Option[T] = buildErrorResult()
175 
176   def buildErrorResult(): Option[EtcdRequestError] = for {
177     status <- statusCodeOp if status >= 400
178     body <- JsonMethods.parse(bytes.toString("UTF-8")).extractOpt[EtcdErrorMessage]
179   } yield EtcdRequestError(EtcdResponseCode(status), headsOp, body)

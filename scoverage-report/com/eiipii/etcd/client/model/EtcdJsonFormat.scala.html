57 case object EtcdListDirNodeSerializer extends CustomSerializer[EtcdListDirNode](
58   implicit format => ({
59     case rootNode: JValue =>
60       (rootNode \ "dir") match {
61         case JBool(true) =>
62           EtcdListDirectory(
63             true,
64             (rootNode \ "key").extract[EtcdKey],
65             (rootNode \ "nodes").extract[List[EtcdListDirNode]],
66             (rootNode \ "modifiedIndex").extract[EtcdIndex],
67             (rootNode \ "createdIndex").extract[EtcdIndex]
68           )
69         case _ => EtcdListValue(
70           false,
71           (rootNode \ "key").extract[EtcdKey],
72           (rootNode \ "value").extract[EtcdValue],
73           (rootNode \ "modifiedIndex").extract[EtcdIndex],
74           (rootNode \ "createdIndex").extract[EtcdIndex]
75         )
76       }
77   }, {
78     case value: EtcdErrorMessage => value match {
79       case EtcdError(cause, errorCode, index, message) =>
80         ("cause", cause) ~
81           ("errorCode", errorCode) ~
82           ("index", index) ~
83           ("message", message)
84       case EtcdSimpleError(message) =>
85         ("message", message)
86     }
87   })
88 )

103       }
104   }, {

117 case object EtcdMemberStatsSerialized extends CustomSerializer[EtcdGetLeaderStatsResponse](
118   implicit format => ({
119     case rootNode: JValue =>
120       val members = (rootNode \ "followers") match {
121         case JObject(members) =>
122           members.map {
123             case (id, json) =>
124               val counts = (json \ "counts").extract[EtcdCounts]
125               val latency = (json \ "latency").extract[EtcdLatency]
126               EtcdMemberStats(
127                 id,
128                 counts,
129                 latency
130               )
131           }
132         case _ => List[EtcdMemberStats]()
133       }
134       EtcdGetLeaderStatsResponse((rootNode \ "leader").extract[String], members)
135   }, {
136     case value: EtcdGetLeaderStatsResponse =>
137       val followersJ: JObject = (JObject() /: value.followers) {
138         case (jsonPart, member) =>
139           jsonPart ~ (
140             member.memberId,
141             ("counts",
142               ("fail", member.counts.fail) ~ ("success", member.counts.success)) ~ ("latency",
143                 ("average", member.latency.average) ~
144                 ("current", member.latency.current) ~
145                 ("maximum", member.latency.maximum) ~
146                 ("minimum", member.latency.minimum) ~
147                 ("standardDeviation", member.latency.standardDeviation))
148           )
149       }
150       ("leader", value.leader) ~ ("followers", followersJ)
151       JObject(List(
152         "leader" -> JString(value.leader),
153         "followers" -> followersJ
154       ))
155   })
156 )

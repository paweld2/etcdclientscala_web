21 /**
22  * Contains methods to define well-formed instances of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]], [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]], hidden nodes and [[com.eiipii.etcd.client.model.EtcdValue EtcdValue]].
23  *
24  */
25 object EtcdModel {
26 
27   /**
28    * Returns an instance of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] from a list of node names.
29    *
30    * The key preserves the order of the list given as input.
31    *
32    * @param path A list of strings not containing "/", which are to be used as node names.
33    */
34 
35   def fromPath(path: List[String]): EtcdKey = {
36     if (path.exists(node => node.contains("/"))) {
37       throw new IllegalArgumentException("node names cannot contain '/'")
38     }
39     //An empty string ("") is prepended to the list so as to get a "/" at the beginning of the key.
40     EtcdKey((("") :: path).mkString("/"))
41   }
42 
43   /**
44    * Returns a list of all node names in order.
45    *
46    * @param key An [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] which is to be split in node names and placed in order into a list.
47    */
48   def toPath(key: EtcdKey): List[String] = {
49     //The first element is dropped, since it is always and empty string ("").
50     key.key.split("/").toList.drop(1)
51   }
52 
53   /**
54    * Creates a well-formed instance of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]]  from a String that represents a path to an etcd node.
55    *
56    * @param key the String representation of the key
57    */
58   def key(key: String): EtcdKey = {
59     if (key == null || key.isEmpty) {
60       throw new IllegalArgumentException("key name can not be null.")
61     }
62     if (!key.startsWith("/")) {
63       throw new IllegalArgumentException(s"key name must start with '/'. Provided value $key")
64     }
65     if (key.endsWith("/")) {
66       throw new IllegalArgumentException(s"key name must not end with '/'. Provided value $key")
67     }
68     EtcdKey(key)
69   }
70 
71   /**
72    * Creates a well-formed instance of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] from a String that represents a path to an etcd node and a String that represents the name of the node.
73    *
74    * @param dir the [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]] which will be appended with a name of a node.
75    * @param key the string representation of the name of the node.
76    */
77   def key(dir: EtcdDirectory, key: String): EtcdKey = EtcdModel.key(dir.dir + key)
78 
79   /**
80    * Creates a well-formed instance of [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]] from a String that represents a path.
81    *
82    * @param dir the string representation of the directory.
83    */
84   def directory(dir: String): EtcdDirectory = {
85     if (dir == null) {
86       throw new IllegalArgumentException("Directory name can not be null.")
87     }
88     if (!dir.startsWith("/")) {
89       throw new IllegalArgumentException(s"Directory name must start with '/'. Provided value $dir")
90     }
91     if (!dir.endsWith("/")) {
92       throw new IllegalArgumentException(s"Directory must end with '/'. Provided value $dir")
93     }
94     EtcdDirectory(dir)
95   }
96   /**
97    * Creates a well-formed instance of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] from a string that represents a path to a node and a string that represents the name of the node.
98    *
99    * For more information on hidden nodes see [[https://coreos.com/etcd/docs/latest/v2/api.html#creating-a-hidden-node Creating a hidden node]].
100    *
101    * @param dir the string representation of the directory.
102    * @param node the string representation of the node's name.
103    * @return A [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] composed of a path and the name of a node prepended with '_'.
104    *
105    */
106   def hiddenNode(dir: String, node: String): EtcdKey = {
107     if (dir == null || dir.isEmpty) {
108       throw new IllegalArgumentException("Directory name can not be null.")
109     }
110     if (!dir.startsWith("/")) {
111       throw new IllegalArgumentException(s"Directory name must start with '/'. Provided value $dir")
112     }
113     if (!dir.endsWith("/")) {
114       throw new IllegalArgumentException(s"Directory must end with '/'. Provided value $dir")
115     }
116     if (node == null || node.isEmpty) {
117       throw new IllegalArgumentException("node name cannot be null.")
118     }
119     if (node.endsWith("/") || node.startsWith("/")) {
120       throw new IllegalArgumentException(s"Node name must not end or start with '/'. Provided value $node")
121     }
122     EtcdKey(dir + "_" + node)
123   }
124 
125   /**
126    * Creates a well-formed instance of [[com.eiipii.etcd.client.model.EtcdValue EtcdValue]]
127    *
128    * @param value the string representation of the value
129    */
130   def value(value: String): EtcdValue = EtcdValue(value)
131 
132   /**
133    * Returns an instance of [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] from a [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]].
134    *
135    * An EtcdDirectory can be treated as an EtcdKey by dropping the closing '/'.
136    *
137    * @param dir the [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]] that is to be treated as a key.
138    */
139   def keyOfDirectory(dir: EtcdDirectory) = EtcdKey(dir.dir.dropRight(1))
140 
141   /**
142    * Returns an instance of [[com.eiipii.etcd.client.model.EtcdDirectory EtcdDirectory]] from a [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]].
143    *
144    * An EtcdDirectory can be treated as an EtcdKey by adding a closing '/'.
145    *
146    * @param key the [[com.eiipii.etcd.client.model.EtcdKey EtcdKey]] that is to be treated as a directory.
147    */
148   def directoryOfKey(key: EtcdKey) = EtcdDirectory(key.key + "/")
149 }

361    *

373 /**
374  * Callback for a wait operation.
375  *
376  * Extends [[com.eiipii.etcd.client.model.EtcdWaitCallback EtcdWaitCallback]].
377  * It should be used as input for [[com.eiipii.etcd.client.EtcdClient.waitForKey EtcdClient.waitForKey]].
378  *
379  * It contains methods for retrieving an [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] received once a change to etcd is registered.
380  *
381  */
382 class FutureBasedEtcdWaitCallback extends EtcdWaitCallback {
383   private[this] val p = Promise[EtcdGetKeyResponse]
384 
385   /**
386    *
387    * @param response [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] from etcd sent when a change on a key or directory takes place.
388    */
389   override def onValueChange(response: EtcdGetKeyResponse): Unit = p.success(response)
390 
391   /**
392    *
393    * @param t Throwable from an error of an [[com.eiipii.etcd.client.EtcdClient.waitForKey EtcdClient.waitForKey]] operation.
394    */
395   override def onError(t: Throwable): Unit = p.failure(t)
396 
397   /**
398    *
399    * @return Future wrapped around the [[com.eiipii.etcd.client.model.EtcdGetKeyResponse EtcdGetKeyResponse]] to be returned once the callback is completed.
400    */
401   def future: Future[EtcdGetKeyResponse] = p.future
402 
403   /**
404    *
405    * @return true if the callback has been completed, false otherwise.
406    */
407   def isCompleted: Boolean = p.isCompleted
408 }
